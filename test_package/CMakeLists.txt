cmake_minimum_required(VERSION 3.25)

project(cbor_tags_test_package LANGUAGES CXX)

find_package(cbor_tags REQUIRED)

add_executable(test_package test_package.cpp)
if(TARGET cbor::cwt_openssl)
  target_link_libraries(test_package PRIVATE cbor::cwt_openssl)
  target_compile_definitions(test_package PRIVATE CBOR_TAGS_TEST_PACKAGE_HAS_CWT_OPENSSL=1)
elseif(TARGET cbor::cwt_wolfssl)
  target_link_libraries(test_package PRIVATE cbor::cwt_wolfssl)
  target_compile_definitions(test_package PRIVATE CBOR_TAGS_TEST_PACKAGE_HAS_CWT_WOLFSSL=1)
else()
  target_link_libraries(test_package PRIVATE cbor::cwt)
endif()
if(CMAKE_CXX_STANDARD)
  set_property(TARGET test_package PROPERTY CXX_STANDARD ${CMAKE_CXX_STANDARD})
endif()
set_property(TARGET test_package PROPERTY CXX_STANDARD_REQUIRED ON)
set_property(TARGET test_package PROPERTY CXX_EXTENSIONS OFF)
